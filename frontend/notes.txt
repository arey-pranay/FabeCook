const registerSubmmit = async () => {
    try {
      setSuccess("");
      setError("");
      setLoading(true);
      const { data } = await axios.post(
        `${process.env.REACT_APP_BACKEND_URL}/register`,
        {
          first_name,
          last_name,
          email,
          password,
          bYear,
          bMonth,
          bDay,
          gender,
        }
      );
      setLoading(false);
      setSuccess(
        "Registered Successfully ! Let's just quickly verify your email"
      );
      setError("");
    } catch (error) {
      setLoading(false);
      setSuccess("");
      setError(error.response.data.message);
    }
  };
